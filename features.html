// Example Rust Rocket trading strategy
#[strategy(name = "Momentum Strategy")]
pub fn momentum_strategy(market_data: &MarketData) -> Signal {
    let short_ma = market_data.calculate_ma(5);
    let long_ma = market_data.calculate_ma(20);
    
    if short_ma > long_ma * 1.02 {
        Signal::Buy {
            price: market_data.current_price,
            size: calculate_position_size(),
        }
    } else if short_ma < long_ma * 0.98 {
        Signal::Sell {
            price: market_data.current_price,
            size: current_position(),
        }
    } else {
        Signal::Hold
    }
}
